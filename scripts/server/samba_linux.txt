# ASSUNTO: Script de instalacao e principais comandos do SAMBA4.
# =========================================================================================================================
#
#   Instalacao de ferramentas de rede:
#
sudo apt install net-tools
#

#   Inicia-se com a instalacao do samba4.
#
sudo apt install samba -y
#
# -----------------------------------------------
# 
#   Agora, e necessario incluir os usuarios do samba.
#   E necessario rodar o comando para cada user a ser criado.
#   Associe tambem uma senha para cada usuario criado.
#
sudo smbpasswd -a usuario
#
# -----------------------------------------------------------
# 
#   Faz-se necessario tambem, intalar alguns adicionais:
#
sudo apt install vim samba-common fuse gvfs gvfs-fuse gvfs-common
#
# -----------------------------------------------------
#
#  Tambem e necessario configurar as pastas a serem compartilhadas no arquivo /etc/samba/smb.conf
#
sudo vim /etc/sama/smb.conf
#
# -----------------------------------------------------
